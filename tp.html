Bienvenue dans la gestion des tâches

Bienvenue sur notre application de gestion des tâches. Cliquez sur les boutons ci-dessous pour interagir avec la page.

        
            
                // Sélection des éléments dans le DOM
                const titre = document.getElementById('titre');
                const paragraphe = document.getElementById('paragraphe');
                const boutonChangerTexte = document.getElementById('changerTexte');
                const boutonChangerCouleur = document.getElementById('changerCouleur');
        
                let changePara = ()=>{
                    paragraphe.textContent = 'Nouveau texte pour le paragraphe';
                }
                let changetitreColor =()=>{
                    const color =["#f58a6e","#7beca4","#51395a","#004c9b"]
                    const randomColor = color[Math.floor(Math.random() * color.length)];
                    titre.style.color = randomColor;
                }
                boutonChangerTexte.addEventListener('click', changePara);
                boutonChangerCouleur.addEventListener('click', changetitreColor);
